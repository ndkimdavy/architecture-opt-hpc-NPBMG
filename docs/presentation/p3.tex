\begin{frame}{Analyse de la qualité du code (CQA)}
    \begin{itemize}
        \fontsize{9}{10}\selectfont
        \item Pourquoi un écart de 65 \% en faveur du Fortran ?
    \end{itemize}
    \begin{block}{\small Diagnostic technique}
        \fontsize{8}{9}\selectfont
        Le compilateur AOCC parvient à vectoriser totalement le Fortran, mais pas en C++.
    \end{block}
    \begin{itemize}
        \fontsize{8}{9}\selectfont
        \item \textbf{Fortran} : Pas d'ambiguïté (aliasing) sur les tableaux $\rightarrow$ Utilisation optimale des unités AVX-512.
        \item \textbf{C++} : Structure de pointeurs complexe $\rightarrow$ L'aliasing empêchant la vectorisation automatique.
    \end{itemize}
    \vspace{0.1cm}
    \begin{alertblock}{Conséquence}
        \fontsize{8}{9}\selectfont
        Le binaire C++ n'utilise pas les unités AVX-512 du Ryzen 7, limitant drastiquement le débit de calcul.
    \end{alertblock}
\end{frame}
